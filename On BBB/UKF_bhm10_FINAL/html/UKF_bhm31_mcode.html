1function [ukfout] = UKF_bhm3(i,Param,z_ukf,state_ukf,w_ukf,est_ukf,P,Q,R,dt,ndim,zdim)

  3%function [P,state_ukf,est_ukf, SOC] = UKF_bhm3(i,Param,z_ukf,state_ukf,w_ukf,est_ukf,P,Q,R,dt,ndim,zdim)
  4%function [t_predict,P,state_ukf,est_ukf] = UKF_bhm(z_ukf,tau_model,state_ukf,w_ukf,est_ukf,P,Q,R,dt,xa0,ndim)
  5
  6% Function expanded to N dimensional state vector
  7% Array indices must begin at 1 and not 0
  8% Sigma points are denoted by the "Chi" variables
  9
 10%%SOC = ones(1,N); 
 11%%state_meas = ones(2,1);
 12
 13qmax = Param.qmax;     
 14Cmax = Param.Cmax;
 15Ccb0 = Param.Ccb0;
 16Ccb1 = Param.Ccb1;
 17Ccb2 = Param.Ccb2;
 18Ccb3 = Param.Ccb3;
 19Rs = Param.Rs;
 20
 21Cs = Param.Cs;
 22Rcp0 = Param.Rcp0;
 23Rcp1 = Param.Rcp1;
 24Rcp2 = Param.Rcp2;
 25Ccp0 = Param.Ccp0;
 26Ccp1 = Param.Ccp1;
 27Ccp2 = Param.Ccp2;
 28
 29Rp = Param.Rp;
 30Jt = Param.Jt;
 31hcp = Param.hcp;
 32hcs = Param.hcs;
 33ha = Param.ha;   %heat transfer coeff
 34Ta = Param.Ta;    %ambient temp degC
 35
 36
 37%SOC(1) = 100;
 38
 39%%x_total(:,1) = x;
 40qb = state_ukf(1,1);
 41qcp = state_ukf(2,1);
 42qcs = state_ukf(3,1);
 43%Tb = state_ukf(4,1); %Removed Tb from state and meas vectors
 44
 45SOC = 1 - (qmax-qb)/Cmax;
 46Cb = Ccb0 + Ccb1*SOC+Ccb2*SOC^2+Ccb3*SOC^3;
 47Ccp = Ccp0 + Ccp1*exp(Ccp2*SOC);
 48Rcp = Rcp0 + Rcp1*exp(Rcp2*SOC);
 49
 50%qhat = qmax - (qmax*0.99 - Cmax)*(1-SOC);
 51%Vb = qhat/Cb; %Diverges
 52
 53Vb = qb/Cb;     %Works with this! Only diverges at very end.
 54Vcs = qcs/Cs;
 55Vcp = qcp/Ccp;
 56Vp = Vb - Vcp - Vcs;
 57
 58ip = Vp/Rp;
 59ib = ip + i;
 60icp = ib - Vcp/Rcp;
 61
 62qbdot = -ib;
 63qcpdot = icp;
 64qcsnew = ib*Rs*Cs;
 65%Tbdot = 1/Jt*(hcp*Vcp^2/Rcp + hcs*Vcs^2/Rs - ha*(Tb-Ta));
 66
 67x = [ qb + qbdot*dt;
 68        qcp + qcpdot*dt;
 69        qcsnew];
 70%{  
 71x = [ qb + qbdot*dt;
 72        qcp + qcpdot*dt;
 73        qcsnew;
 74        Tb + Tbdot*dt];    
 75%}
 76        
 77state_model = x + w_ukf;   %voltage and tau
 78state_ukf(:,2)= state_model;
 79xa = est_ukf(:,1);
 80%% Selection of Sigma Points %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 81%W0 = -0.8;  % -1 < W0 < 1
 82%W0 = 0.99;              %Weights for tau are different than for voltage
 83%W1 = (1-W0)/(2*2);
 84W = zeros(2*ndim+1,1);
 85W(1) = 0.99;                            %Sigma point array begins at 1 and not 0
 86SQRTM_P = real(sqrtm( 2*P/(1-W(1)) ));  %Square root of matrix is converted to real type
 87Chi = zeros(ndim,2*ndim+1);
 88Chi(:,1) = xa; % xa is old state estimate feed back from algorithm
 89
 90for k = 2:ndim+1            %Generate sigma points with +sqrt. Increase indices by 1 since arrays don't start from 0
 91Chi(:,k) = xa + SQRTM_P(:,k-1); 
 92W(k) = (1-W(1))/(2*2);
 93
 94Chi(:,ndim+k) = xa - SQRTM_P(:,k-1); 
 95W(ndim+k) = (1-W(1))/(2*2);
 96end
 97%% Model Forecast Step %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 98% Propagate Sigma Points through non-linear transform
 99ChiF = zeros(ndim,2*ndim+1);
100
101for k = 1:2*ndim+1          %Forecast sigma points
102ip = Vp/Rp;
103ib = ip + i;        
104Chi_qcsnew = ib*Rs*Cs;    %%Consider finding qcsdot
105
106ChiF(:,k) = [ Chi(1,k) + qbdot*dt;    %Symbol for sigma pt is chi
107        Chi(2,k) + qcpdot*dt;
108        Chi_qcsnew];            %%% Not sure about this!!!
109
110%{
111ChiF(:,k) = [ xa(1) + qbdot*dt;    %Symbol for sigma pt is chi
112        xa(2) + qcpdot*dt;
113        Chi_qcsnew;            %%% Not sure about this!!!
114        xa(4) + Tbdot*dt];
115%}
116
117%a = (1/(1+dt/tau_model))*Chi(1,k);
118%tau_meas = (1/(-1+Chi(1,1)/a))*dt;  %There is an issue forecasting tau.  To work better, need 2 points in history
119%ChiF(:,k) = [a; tau_model]+w_ukf;
120%ChiF(:,k) = [a; tau_meas];
121end
122
123%% Compute the mean and covariance of forecast
124ChiF_mean = zeros(ndim,1);
125
126for k = 1:2*ndim+1          %Calculate mean of sigma points
127ChiF_mean = ChiF_mean + W(k)*ChiF(:,k); 
128end
129
130PX = zeros(ndim); 
131%PX = zeros(ndim,ndim);      
132for k = 1:2*ndim+1    %Calculate covariance matrix of sigma pts
133PX = PX + W(k)*(ChiF(:,k) - ChiF_mean)*(ChiF(:,k) - ChiF_mean)';
134end
135
136PX = PX+Q;
137%% Propagate the sigma points through the observation model
138%ZF = ChiF;                  %Expected measurements based on forecasted sigma points
139ZF = [1/Cb -1/Ccp -1/Param.Cs]*ChiF; %Expected measurements based on forecasted sigma points
140
141%% Compute mean and covariance of transformed observations
142ZF_mean = zeros(zdim,1);
143%ZF_mean = zeros(ndim,1);
144
145for k = 1:2*ndim+1    %Calculate mean of expected measurements
146    ZF_mean = ZF_mean + W(k)*ZF(:,k); 
147end
148
149PZ = zeros(zdim);
150%PZ = zeros(ndim,ndim);
151for k = 1:2*ndim+1    %Covar of expected measurements
152PZ = PZ + W(k)*(ZF(:,k) - ZF_mean)*(ZF(:,k) - ZF_mean)';
153end
154PZ = PZ+R;
155%% Compute the cross covariance between XF and Zf
156PXZ = zeros(ndim,zdim);
157%PXZ = zeros(ndim,ndim);
158
159for k = 1:2*ndim+1    %Cross covariance of sigma points and expected measurements
160PXZ = PXZ + W(k)*(ChiF(:,k) - ChiF_mean)*(ZF(:,k) - ZF_mean)'; 
161end
162%% Data Assimilation Step %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
163%K = PXZ*pinv(PZ);
164K = PXZ*PZ^(-1);
165xa = ChiF_mean + K*(z_ukf - ZF_mean);
166P = PX - K*PZ*K';
167est_ukf(:,2) = xa;
168%t_predict = -xa(2,1)*log(0.5/xa0);
169ukfout = struct('P',P,'state_ukf',state_ukf,'est_ukf',est_ukf,'SOC',SOC);
170
171end
